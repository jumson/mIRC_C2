alias update {
  showlog $chan 500
}

//ensure the $script_dir variable is set already in var
// something: %script_dir "C:\Users\munso\Desktop\script"
ON 1:CONNECT: {
  load -rs %script_dir $+ \mIRC_C2_Box.mrc
  load -rs %script_dir $+ \c2Report-5ws.mrc
  load -rs %script_dir $+ \c2Report-9line.mrc
  load -rs %script_dir $+ \c2Report-spot.mrc
  load -rs %script_dir $+ \c2Report-clears.mrc
}

ON 1:JOIN:*: {
  IF ($nick == $me) {
    update
  }
}

alias beeps {
  while ($dialog(mc2).active == $false) {
    beep
  }
}

alias closeit {
  echo -s $fopen(aguy)
  if ($fopen(aguy) == aguy) {
    echo aguy is open!!!
    fclose aguy
    echo -s aguy is closed
  }
}

alias restart {
  unload -rs %script_dir $+ \mIRC_C2_Box.mrc
  unload -rs %script_dir $+ \c2Report-5ws.mrc
  unload -rs %script_dir $+ \c2Report-9line.mrc
  unload -rs %script_dir $+ \c2Report-spot.mrc
  unload -rs %script_dir $+ \c2Report-clears.mrc  
  load -rs %script_dir $+ \mIRC_C2_Box.mrc
  load -rs %script_dir $+ \c2Report-5ws.mrc
  load -rs %script_dir $+ \c2Report-9line.mrc
  load -rs %script_dir $+ \c2Report-spot.mrc
  load -rs %script_dir $+ \c2Report-clears.mrc
}
